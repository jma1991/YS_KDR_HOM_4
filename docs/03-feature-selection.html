modelGeneVar

Identify highly variable genes:

dec <- metadata(sce)$modelGeneVar

hvg <- getTopHVGs(dec, var.field = "bio", var.threshold = 0, fdr.field = "FDR", fdr.threshold = 0.05)

hvg <- setdiff(hvg, del)

metadata(sce)$modelGeneVar$hvg <- sel
